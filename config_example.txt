# Configuración del Bot de Monitoreo de Red
# Copia estas líneas a tu archivo .env y configura los valores según tu entorno

# ==================== TELEGRAM BOT ====================
# Token del bot de Telegram (obtenido de @BotFather)
TELEGRAM_TOKEN=your_telegram_bot_token_here

# ==================== MQTT BROKER ====================
# Configuración básica del broker MQTT
MQTT_BROKER=localhost
MQTT_PORT=1883
MQTT_TOPIC=network/monitoring

# ==================== SEGURIDAD MQTT ====================
# Autenticación (opcional pero recomendado)
MQTT_USERNAME=tu_usuario_mqtt
MQTT_PASSWORD=tu_password_mqtt

# SSL/TLS (recomendado para producción)
MQTT_USE_SSL=false
MQTT_CA_CERT=/path/to/ca.crt
MQTT_CERT_FILE=/path/to/client.crt
MQTT_KEY_FILE=/path/to/client.key

# Configuración de conexión MQTT
MQTT_KEEPALIVE=60
MQTT_CONNECT_TIMEOUT=10
MQTT_MAX_RETRIES=3

# ==================== MONITOREO ====================
# Configuración de parámetros de ping y monitoreo
PING_COUNT=4
MONITOR_INTERVAL=10

# Configuración de timeouts para comandos de red  
PING_TIMEOUT=15
MAX_PING_RETRIES=2

# ==================== LOGGING ====================
# Configuración del sistema de logging

# Directorio donde se guardarán los logs
LOGS_DIRECTORY=logs

# Nivel de logging (DEBUG, INFO, WARNING, ERROR, CRITICAL)
LOG_LEVEL=INFO

# Días para retener logs antes de eliminarlos automáticamente
LOG_RETENTION_DAYS=30

# Habilitar logs de debug más detallados (true/false)
ENABLE_DEBUG_LOGS=false
